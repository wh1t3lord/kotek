cmake_minimum_required(VERSION 3.19.3)

project(kotek.core.defines.static.render.gl VERSION 1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if ("${KOTEK_OPENGL}" STREQUAL "")
	set(KOTEK_OPENGL "ON" CACHE INTERNAL "")
endif()

if ("${KOTEK_OPENGL}" STREQUAL "ON")
	list(APPEND KOTEK_OUTPUT_FLAGS -DKOTEK_OPENGL)

	# TODO: think about supporting modern version and split two things, like if we specify for legacy version 3 it means will be added all project until modern
	# So if legacy = 1 it means from 1 version will be add to modern
	set(KOTEK_OPENGL_MODERN_VERSION "3")
	set(KOTEK_OPENGL_LEGACY_VERSION "3")


	## OPENGL ES ##
	if ("${KOTEK_DEF_RENDER_GL_ES_INDIRECT_DRAWING_TOTAL_COMMANDS}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_ES_INDIRECT_DRAWING_TOTAL_COMMANDS 50)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT 8)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_REALLOCATION_ENABLED}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_REALLOCATION_ENABLED 1)
	endif()



	## OPENGL ##

	if ("${KOTEK_DEF_RENDER_GL_INDIRECT_DRAWING_TOTAL_COMMANDS}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_INDIRECT_DRAWING_TOTAL_COMMANDS 50)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT 8)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_REALLOCATION_ENABLED}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_REALLOCATION_ENABLED 1)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_BINDLESS_TEXTURE_ENABLED}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_BINDLESS_TEXTURE_ENABLED 1)
	endif()

	if ("${KOTEK_DEF_RENDER_GL_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT}" STREQUAL "")
		set(KOTEK_DEF_RENDER_GL_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT 32)
	endif()

endif()

file(WRITE ${KOTEK_FILENAME_FOR_GENERATING_PREPROCESSORS} "#pragma once

#ifdef KOTEK_OPENGL
	#define KOTEK_USE_OPENGL
	#define KOTEK_USE_OPENGL_MODERN_VERSION ${KOTEK_OPENGL_MODERN_VERSION}
	#define KOTEK_USE_OPENGL_LEGACY_VERSION ${KOTEK_OPENGL_LEGACY_VERSION}

	#define KOTEK_DEF_RENDER_GL_ES_INDIRECT_DRAWING_TOTAL_COMMANDS ${KOTEK_DEF_RENDER_GL_ES_INDIRECT_DRAWING_TOTAL_COMMANDS}
	#define KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT ${KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}
	#define KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_REALLOCATION_ENABLED ${KOTEK_DEF_RENDER_GL_ES_GEOMETRY_MANAGER_REALLOCATION_ENABLED}

	#define KOTEK_DEF_RENDER_GL_INDIRECT_DRAWING_TOTAL_COMMANDS ${KOTEK_DEF_RENDER_GL_INDIRECT_DRAWING_TOTAL_COMMANDS}
	#define KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT ${KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}
	#define KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_REALLOCATION_ENABLED ${KOTEK_DEF_RENDER_GL_GEOMETRY_MANAGER_REALLOCATION_ENABLED}
	#define KOTEK_DEF_RENDER_GL_BINDLESS_TEXTURE_ENABLED ${KOTEK_DEF_RENDER_GL_BINDLESS_TEXTURE_ENABLED}

	// render graph simplified
	#define KOTEK_DEF_RENDER_GL_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT ${KOTEK_DEF_RENDER_GL_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT}
#endif
")

add_library(${PROJECT_NAME} STATIC
	"include/kotek_std_preprocessors.h"
	"include/kotek_core_defines_static_render_gl.h"
	"src/main_core_defines_static_render_gl_dll.cpp")
	
set_target_properties(${PROJECT_NAME} PROPERTIES FOLDER "framework/core/macros/independent")
target_compile_definitions(${PROJECT_NAME} PUBLIC ${KOTEK_OUTPUT_FLAGS})

target_link_libraries(${PROJECT_NAME} PUBLIC kotek.core.defines.static.cpp)
