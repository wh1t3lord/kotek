cmake_minimum_required(VERSION 3.19.3)

project(kotek.core.defines.static.render.bgfx VERSION 1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if ("${KOTEK_BGFX}" STREQUAL "")
	set(KOTEK_BGFX "ON" CACHE INTERNAL "")
endif()

if ("${KOTEK_BGFX}" STREQUAL "ON")
	list(APPEND KOTEK_OUTPUT_FLAGS -DKOTEK_BGFX)

	if ("${KOTEK_DEF_RENDER_BGFX_INDIRECT_DRAWING_TOTAL_COMMANDS}" STREQUAL "")
		set(KOTEK_DEF_RENDER_BGFX_INDIRECT_DRAWING_TOTAL_COMMANDS 50)
	endif()

	if ("${KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}" STREQUAL "")
		set(KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT 8)
	endif()

	if ("${KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_REALLOCATION_ENABLED}" STREQUAL "")
		set(KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_REALLOCATION_ENABLED 1)
	endif()

	if ("${KOTEK_DEF_RENDER_BGFX_BINDLESS_TEXTURE_ENABLED}" STREQUAL "")
		set(KOTEK_DEF_RENDER_BGFX_BINDLESS_TEXTURE_ENABLED 1)
	endif()

	if ("${KOTEK_DEF_RENDER_BGFX_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT}" STREQUAL "")
		set(KOTEK_DEF_RENDER_BGFX_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT 32)
	endif()

endif()

message("[${PROJECT_NAME}]: BGFX is '${KOTEK_BGFX}'")

file(WRITE ${KOTEK_FILENAME_FOR_GENERATING_PREPROCESSORS} "#pragma once

#ifdef KOTEK_BGFX
	#define KOTEK_USE_BGFX

	#define KOTEK_DEF_RENDER_BGFX_INDIRECT_DRAWING_TOTAL_COMMANDS ${KOTEK_DEF_RENDER_BGFX_INDIRECT_DRAWING_TOTAL_COMMANDS}
	#define KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT ${KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_MAX_BUFFERS_COUNT}
	#define KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_REALLOCATION_ENABLED ${KOTEK_DEF_RENDER_BGFX_GEOMETRY_MANAGER_REALLOCATION_ENABLED}
	#define KOTEK_DEF_RENDER_BGFX_BINDLESS_TEXTURE_ENABLED ${KOTEK_DEF_RENDER_BGFX_BINDLESS_TEXTURE_ENABLED}

	// render graph simplified
	#define KOTEK_DEF_RENDER_BGFX_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT ${KOTEK_DEF_RENDER_BGFX_RENDER_GRAPH_SIMPLIFIED_MAX_PASS_COUNT}


#endif
")

add_library(${PROJECT_NAME} STATIC
	"include/kotek_std_preprocessors.h"
	"include/kotek_core_defines_static_render_bgfx.h"
	"src/main_core_defines_static_render_bgfx_dll.cpp")
	
set_target_properties(${PROJECT_NAME} PROPERTIES FOLDER "framework/core/macros/independent")
target_compile_definitions(${PROJECT_NAME} PUBLIC ${KOTEK_OUTPUT_FLAGS})

target_link_libraries(${PROJECT_NAME} PUBLIC kotek.core.defines.static.cpp)
